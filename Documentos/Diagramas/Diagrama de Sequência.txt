@startuml
actor Cliente
entity "Sistema" as Sistema
entity "Carrinho" as Carrinho
entity "Banco de Dados" as BD

Cliente -> Carrinho : acessa carrinho
Carrinho -> Sistema : valida itens no carrinho
Sistema -> BD : verifica disponibilidade dos produtos
Cliente -> Sistema : escolhe forma de pagamento
Sistema -> BD : registra compra e cria pedido
Sistema -> Cliente : confirma compra

@enduml
